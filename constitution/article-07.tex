\Article{A-Board Structure, Responsibilities, and Roles}

% SECTION A: WEB ADMIN
\Section{Web Admin}

\begin{enumerate}
  \item This position is appointed by a vote of the incoming E-Board. 
  \item Responsible for maintaining all RITSEC websites.
\end{enumerate}

% SECTION B: JUNIOR TECH LEAD
\Section{Junior Tech Lead}

\begin{enumerate}
  \item This position is appointed by a vote of the incoming E-Board.
  \item Responsible for assisting the Tech Lead.
\end{enumerate}

% SECTION C: JUNIOR OPERATIONS LEAD
\Section{Junior Operations Lead}

\begin{enumerate}
  \item This position is appointed by a vote of the incoming E-Board.
  \item Responsible for assisting the Operations Lead.
\end{enumerate}
